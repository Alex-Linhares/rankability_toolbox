% Ranking matrix R; R(i,j)=percentage of ranking vectors in P that have
% item i in jth rank position
n=size(D,1);
R=zeros(n,n);
for i=1:n
for j=1:p
R(P(i,j),i)=R(P(i,j),i)+1;
end
end
